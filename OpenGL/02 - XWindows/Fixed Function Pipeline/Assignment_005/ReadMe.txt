///		1. SHOW SPINNING 2D TRIANGLE AND ROLLING 2D RECTANGLE 
///
/////////////////////////////////////////////////////////////////////////////////////////////////////

1. Globally : NO CHANGE.

2. main()  : NO CHANGE.
						
3. Resize() : NO CHANGE.

4. display() : Changes are as follows :
				1. --> void display(void)
						{
							//variables
							static GLfloat tAngle=0.0f;
							static GLfloat rAngle=0.0f;
							.
							.

							// 2D TRIANGLE
							glLoadIdentity();
							glTranslatef(-1.5f,0.0f,-5.0f);
							
							glRotatef(tAngle, 0.0f,1.0f,0.0f);	//Y-AXIS SPINNING
							.
							.

							// 2D RECTANGLE
							glLoadIdentity();
							glTranslatef(1.5f,0.0f,-5.0f);

							glRotatef(rAngle, 1.0f,0.0f,0.0f);	//X-AXIS ROLLING
							.
							.
														
							tAngle+=0.05f;
							rAngle+=0.05f;
							
							if(tAngle >=360.0f)
								tAngle=0.0f;
								
							if(rAngle >=360.0f)
								rAngle=0.0f;

						}
